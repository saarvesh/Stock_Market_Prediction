def to_words(content):
    letters_only= re.sub("[^a-zA-Z']", " ", content)
    words=letters_only.lower().split()
    stops=set(stopwords.words("english"))
    meaningful_words = [w for w in words if w not in stops]
    return (" ".join(meaningful_words))
